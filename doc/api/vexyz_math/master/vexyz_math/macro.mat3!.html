macro_rules! mat3 {
    ($s:expr) => {{
        let s = $s as f64;
        Mat3::new(
            vec3!(s, 0, 0),
            vec3!(0, s, 0),
            vec3!(0, 0, s),
        )
    }};
    ($col0:expr, $col1:expr, $col2:expr) => {{
        Mat3::new($col0, $col1, $col2)
    }};
    ($col0:expr, $col1:expr, $col2:expr,) => {{
        Mat3::new($col0, $col1, $col2)
    }};
    ($m00:expr, $m01:expr, $m02:expr,
     $m10:expr, $m11:expr, $m12:expr,
     $m20:expr, $m21:expr, $m22:expr) => {{

            vec3!($m10 as f64, $m11 as f64, $m12 as f64),
            vec3!($m20 as f64, $m21 as f64, $m22 as f64),
        )

Builder macro for creating new 3x3 column major matrices.

Examples

Create a new column major matrix with 0.5 on diagonal:

#[macro_use] extern crate vexyz_math;
use vexyz_math::*;

let a = mat3!(0.5);
let b = Mat3::new(
    vec3!(0.5, 0.0, 0.0),
    vec3!(0.0, 0.5, 0.0),
    vec3!(0.0, 0.0, 0.5),
);
assert_eq!(a, b);

Create a new column major matrix with column0 = vec3(0.5, 0.6, 0.7), column1 = vec3(1.5, 1.6, 1.7), column2 = vec3(2.5, 2.6, 2.7):

#[macro_use] extern crate vexyz_math;
use vexyz_math::*;

let a = mat3!(
    vec3!(0.5, 0.6, 0.7),
    vec3!(1.5, 1.6, 1.7),
    vec3!(2.5, 2.6, 2.7),
);
let b = Mat3::new(
    vec3!(0.5, 0.6, 0.7),
    vec3!(1.5, 1.6, 1.7),
    vec3!(2.5, 2.6, 2.7),
);
assert_eq!(a, b);
